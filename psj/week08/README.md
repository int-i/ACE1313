## 1. Convert if statements in the below code to a switch statement.

```cpp
#include <iostream>

using namespace std;

int main()
{
    char c;
    cin >> c;
    if (c == 'A')
    {
        cout << "Your score is A" << endl;
    }
    else if (c == 'B')
    {
        cout << "Your score is B" << endl;
    }
    else if (c == 'C')
    {
        cout << "Your score is C" << endl;
    }
    else if (c == 'D')
    {
        cout << "Your score is D" << endl;
    }
    return 0;
}
```
